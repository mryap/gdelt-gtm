SELECT
  ActionGeo_CountryCode,
  EventRootCode,
  MonthYear,
  COUNT(1) AS c,
  #RATIO_TO_REPORT(c) OVER(PARTITION BY MonthYear ORDER BY c DESC) norm
FROM
  [gdelt-bq:full.events]
WHERE
  ActionGeo_CountryCode='GT' AND EventRootCode='14'
  GROUP BY
  ActionGeo_CountryCode,
  EventRootCode,
  MonthYear
ORDER BY
  #ActionGeo_CountryCode,
  #EventRootCode,
  MonthYear DESC;
~~~
